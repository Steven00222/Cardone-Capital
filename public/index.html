async function login() { const email = document.getElementById('login-email').value; const password = document.getElementById('login-password').value; const res = await fetch(`${WORKER_URL}/api/login`, { method:'POST', headers:{'Content-Type':'application/json'}, body: JSON.stringify({email,password}) }); if(res.ok){ sessionStorage.setItem('user', email); window.location.href = 'dashboard.html'; } else { alert(await res.text()); } } async function register() { const email = document.getElementById('register-email').value; const password = document.getElementById('register-password').value; const res = await fetch(`${WORKER_URL}/api/register`, { method:'POST', headers:{'Content-Type':'application/json'}, body: JSON.stringify({email,password}) }); if(res.ok){ alert('Registered! You can now login.'); } else { alert(await res.text()); } }
